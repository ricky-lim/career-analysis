    \section{Goal:}\label{goal}

\begin{itemize}
\tightlist
\item
  What is the mean salary value for a researcher (onderzoeker) in The
  Netherlands ?
\item
  What is the gap-ratio on the mean salary value between private and
  public sector on average in The Netherlands ?
\end{itemize}

\begin{itemize}
\tightlist
\item
  \textbf{Web-scraping} Glassdoor-specific query using \texttt{selenium}
  and \texttt{beautifulsoup\ (bs4)} python libraries
\item
  \textbf{Extraction, Transformation, Loading (ETL)} salary data into
  DataFrame using \texttt{pandas} library
\item
  \textbf{Visualization} using\texttt{bokeh}, and \textbf{descriptive
  statistics} on salary data using \texttt{pandas}
\end{itemize}
